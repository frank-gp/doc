```sh
sudo vim /etc/nginx/conf.d/giomr.site.conf

```

```nginx
server {
    listen 80;
    server_name giomr.site;

    root /var/www/giomr.site/html;
    index index.html index.htm;

    location / {
        try_files $uri $uri/ =404;
    }
}
```
