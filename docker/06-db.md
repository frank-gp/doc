### PostgreSQL

```bash
# Create PostgreSQL container
docker run --name postgres \
  -e POSTGRES_PASSWORD=admin \
  -e POSTGRES_DB=postgres_db \
  -d -p 5433:5432 postgres

# PostgreSQL with custom network
docker run --name fgp_db_container \
  -e POSTGRES_PASSWORD=admin \
  -e POSTGRES_DB=postgres_db \
  --network fgp_network \
  -d postgres

# Connect to PostgreSQL from host
psql -h localhost -p 5433 -U postgres

# Connect to PostgreSQL from inside container
docker exec -it postgres psql -U postgres
```

### MongoDB

```bash
# Create basic MongoDB container
docker create --name monguito mongo
docker start monguito

# MongoDB with mapped ports
docker create -p 27017:27017 --name monguito mongo
docker start monguito

# MongoDB with authentication
docker run -p 27017:27017 --name monguito \
  -e MONGO_INITDB_ROOT_USERNAME=nico \
  -e MONGO_INITDB_ROOT_PASSWORD=password \
  -d mongo

# MongoDB with custom network
docker create -p 27018:27017 --name monguito \
  --network my_net \
  -e MONGO_INITDB_ROOT_USERNAME=nico \
  -e MONGO_INITDB_ROOT_PASSWORD=password \
  mongo

# MongoDB connection string
# mongodb://nico:password@localhost:27018/test?authSource=admin
```

### MySQL

```bash
# Download MySQL image
docker pull mysql
docker pull --platform linux/x86_64 mysql  # For specific architectures

# Create MySQL container
docker run --name mysql-container \
  -e MYSQL_ROOT_PASSWORD=admin \
  -e MYSQL_DATABASE=myapp_db \
  -d -p 3306:3306 mysql:8.0
```
